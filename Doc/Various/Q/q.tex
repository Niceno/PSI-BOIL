\section{Visualisation of coherent structures}

We propose the following (see: http://tinyurl.com/2oxj28)

\be
  Q = \frac{1}{2} \left( \Omega_{ij} \Omega_{ij} - S_{ij} S_{ij} \right)
\ee

where:

\be
  \Omega_{ij} = \frac{1}{2} \left( u_{i,j} - u_{j,i} \right),
\ee

and:

\be
  S_{ij} = \frac{1}{2} \left( u_{i,j} + u_{j,i} \right).
\ee

Remember inner Product: Let ${\bf S}$ and ${\bf T}$ be two second order tensors.  The inner
product of ${\bf S}$ and ${\bf T}$ is a scalar, denoted by ${\bf S} : {\bf T}$.
Represent ${\bf S}$ and ${\bf T}$
by their components in a basis.  Then
%
\bea
  {\bf S}:{\bf T} = S_{ij} T_{ij}
                  & = & S_{11} T_{11} + S_{12} T_{12} + S_{13} T_{13} \\
                  & + & S_{21} T_{21} + S_{22} T_{22} + S_{23} T_{23} \\
                  & + & S_{31} T_{31} + S_{32} T_{32} + S_{33} T_{33}
\eea
%
Observe that ${\bf S}:{\bf T} = {\bf T}:{\bf S}$,

\be
  S_{ij}
  =
  \frac{1}{2}
  \left[ \ba{ccc}
           2 \frac{\p u}{\p x}  &  \frac{\p u}{\p y}+\frac{\p v}{\p x}  &  \frac{\p u}{\p z}+\frac{\p w}{\p x}  \\
           \frac{\p v}{\p x}+\frac{\p u}{\p y}  &  2 \frac{\p v}{\p y}  &  \frac{\p v}{\p z}+\frac{\p w}{\p y}  \\
           \frac{\p w}{\p x}+\frac{\p u}{\p z}  &  \frac{\p w}{\p y}+\frac{\p v}{\p z}  &  2 \frac{\p w}{\p z} 
         \ea
  \right]
\ee

\be
  \Omega_{ij}
  =
  \frac{1}{2}
  \left[ \ba{ccc}
           0  &  \frac{\p u}{\p y}-\frac{\p v}{\p x}  &  \frac{\p u}{\p z}-\frac{\p w}{\p x}  \\
           \frac{\p v}{\p x}-\frac{\p u}{\p y}  &  0  &  \frac{\p v}{\p z}-\frac{\p w}{\p y}  \\
           \frac{\p w}{\p x}-\frac{\p u}{\p z}  &  \frac{\p w}{\p y}-\frac{\p v}{\p z}  &  0 
         \ea
  \right]
\ee
